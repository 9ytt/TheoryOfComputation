\begin{enumerate}[\large\protect\expandafter\fbox 1]
  \item 次の各問いに答えよ。解答欄に結果のみ記入せよ。
  \begin{enumerate}[(1)]
    \item $a^2b+ab^2+a+b-ab-1$を因数分解せよ。
    \item 72の正の約数はいくつあるか。
    \item 2次関数$y=2x^2-8x+k$の最小値が$-6$となるように定数$k$の値を定めよ。
    \item 2次不等式$ax^2+bx+6>0$の解が$-2<x<3$であるとき定数$a$, $b$の
      値を求めよ。
    \item 数列\suuretu{a_n}の初項から第$n$項までの和$S_n$が
      $S_n=2^n-n-1$で与えられれているとき，一般項$a_n$を求めよ。
  \end{enumerate}
\vfill

  \item 男女6名ずつ12名のサークルで4名の委員をくじで選ぶことになった。
  \begin{enumerate}[(1)]
    \item 男女同数となる選び方は何通りあるか。
    \item 女子が少なくとも1人選ばれる確率を求めよ。
    \item 女子の方が多く選ばれる確率を求めよ。
  \end{enumerate}
\vfill

  \item 2次方程式$x^2-(k-1)x+k^2-2=0$について次の問いに答えよ。
    \begin{enumerate}[(1)]
      \item 1つの解が$k$であるとき，定数$k$の値を求めよ。
      \item (1)のとき，他の解を求めよ。
      \item 異なる2つの実数解を持つような定数$k$の値の範囲を求めよ。
    \end{enumerate}
\vfill

  \item 
    \begin{caprm}
      四角形ABCDにおいて
      \[ AB=2,~BC=\sqrt3+1,~CD=\sqrt2,~\kaku{ABC}=60\Deg,~\kaku{BCD}=75\Deg \]
      とする。
    \end{caprm}
    \begin{edaenumerate}<3>[(1)]
      \item ACの長さを求めよ。
      \item \kaku{ACB}の大きさを求めよ。
      \item 四角形ABCDの面積を求めよ。
    \end{edaenumerate}
\vfill

\item 2直線を
  \begin{align*}
    l_1 &: 2x-3y+6=0\\
    l_2 &: 3x-y-5=0~とする。
  \end{align*}
  $l_1$と$l_2$の交点をAとするとき，次の問いに答えよ。
  \begin{enumerate}[(1)]
    \item Aを通り，$l_1$と垂直な直線$l_3$の方程式を求めよ。
    \item Aを通り，$x$軸の正の方向と$120\Deg$の角をなす直線$l_4$の
      方程式を求めよ。
    \item Aを中心として半径2の円の方程式を求めよ。
  \end{enumerate}
\vfill
