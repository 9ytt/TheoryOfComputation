        \bm{k=-2}, ~\bm{1}
      \end{gather*}

      \begin{gather*}
        x^2+3x+2=0\\

      \end{gather*}
    
    $k=1$のとき，与式は
      \begin{gather*}
        x^2-1=0\\
        x=\pm1
      \end{gather*}

        \bm{\bunsuu{-1-2\sqrt{7}}{3}<k<\bunsuu{-1+2\sqrt{7}}{3}}
